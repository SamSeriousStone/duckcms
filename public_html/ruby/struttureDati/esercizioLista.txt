Implementa la classe List con i seguenti metodi e attributi:

empty?			#restituisce se (true o false) la lista è vuota

private last_node       #restituisce l'ultimo nodo della lista

last                    #restituisce il valore dell'ultimo nodo

@first_node             #è il primo nodo della lista, non serve un metodo

first                   #restituisce il valore del primo nodo

push(value)		#aggiunge un nodo (passando un valore) in fondo alla lista

to_s			#restituisce una stringa coi valori dei nodi

length                  #restituisce la lunghezza della lista

to_a                    #restituisce un array coi valori della lista

private get_node(n)     #restituisce l'ennesimo nodo della lista

get(n)			#restituisce l'ennesimo valore della lista

delete_at(n)		#rimuove l'ennesimo nodo dalla lista

insert(n,value)	        #inserisce un nodo con un certo valore in posizione n

find_index(value)       #restituisce la prima posizione del valore cercato

include?(value)         #restituisce true se il valore è presente, false altrimenti
